## 위상정렬

1. DFS 방법 (c++)
```
#include<stdio.h>
#include<vector>
#include<stack>
using namespace std;
int N, M;
vector<int>edge[32001];
int visited[32001];
stack<int>stk;

void dfs(int here) {
	visited[here] = 1;

	for (int there : edge[here]) {
		if (!visited[there]) {
			dfs(there);
		}
	}
	stk.push(here);
}

int main() {
	scanf("%d%d", &N, &M);
	for (int i = 0; i < M; i++) {
		int a, b;
		scanf("%d%d", &a, &b);
		edge[a].push_back(b);
	}

	for (int i = 1; i < N+1; i++) {
		if (visited[i] == 0) {
			dfs(i);
		}
	}
	while (1) {
		printf("%d", stk.top());
		stk.pop();
		if (stk.size() == 0) {
			puts("");
			break;
		}
		else printf(" ");
	}
}
```

2. indegree 방법 (c++)
```
#include<stdio.h>
#include<vector>
#include<queue>
using namespace std;

int n, m;
int ind[32005];
vector<int>arr[32005];
queue<int>q;

int a,b;

int main()
{
	scanf("%d%d", &n, &m);
	for (int i = 0; i < m; i++)
	{
		scanf("%d %d", &a, &b);
		arr[a].push_back(b);
		ind[b]++;
	}

	for (int i = 1; i <= n; i++)
	{
		if (ind[i] == 0)
		{
			q.push(i);
		}
	}

	int first = 0;
	while (!q.empty())
	{
		int turn = q.front();
		q.pop();
		if (first)
			printf(" %d", turn);
		else
		{
			printf("%d", turn);
			first++;
		}

		for (int i = 0; i < arr[turn].size(); i++)
		{
			int y = arr[turn][i];
			ind[y]--;
			if (ind[y] == 0)
			{
				q.push(y);
			}
		}
	}
	puts(" ");
}
```
